## Install

```bash
npm install --save react-search-box
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-search-box'
import 'react-search-box/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
